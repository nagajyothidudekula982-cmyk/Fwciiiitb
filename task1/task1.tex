\section*{Section A}

\begin{enumerate}[label=\textbf{Q\arabic*.}]

\item If
\[
A=
\begin{pmatrix}
\cos\theta & \sin\theta\\
-\sin\theta & \cos\theta
\end{pmatrix},
\]
show that
\[
A^TA=I.
\]

\item If
\[
|A|=5,
\]
find the value of
\[
|3A|.
\]

\item Find the value of $k$ such that the system of equations has a unique solution:
\[
x+y+z=6
\]
\[
x+2y+3z=10
\]
\[
2x+3y+kz=15
\]

\item Write the sum of intercepts cut off by the plane
\[
\vec r \cdot (2\hat i+5\hat j+\hat k)=10
\]
on the coordinate axes.

\item If
\[
\vec a = 4\hat i+\hat j+\hat k
\]
and
\[
\vec b = 2\hat i+2\hat j+\hat k,
\]
find a unit vector parallel to
\[
\vec a+\vec b.
\]

\item Solve:
\[
\tan^{-1}x+\tan^{-1}(x-1)=\frac{\pi}{4}
\]

\item Prove that
\[
\tan^{-1}\left(\frac{x+y}{1-xy}\right)
=
\tan^{-1}x+\tan^{-1}y
\]

\item If
\[
f(x)=
\begin{cases}
a\sin x+1, & x<0\\
2, & x=0\\
bx-1, & x>0
\end{cases}
\]
is continuous at $x=0$, find $a$ and $b$.

\item Find
\[
\frac{dy}{dx}
\]
if
\[
y=\sin^{-1}\left(\frac{1-x^2}{1+x^2}\right)
\]

\item Find equations of tangents to
\[
y=x^2
\]
which are perpendicular to
\[
x+2y=5
\]

\item Evaluate:
\[
\int \frac{x^2}{1+5x^2}\,dx
\]

\item Find the particular solution of
\[
\frac{dy}{dx}\cos x = 1+y\sin x
\]
given that
\[
y=1 \quad \text{when } x=0
\]

\item Show that the points
\[
(1,2,-1),\ (2,1,3),\ (4,5,-2),\ (6,3,-1)
\]
are coplanar.

\item A bag contains 3 white balls and 2 black balls while another bag contains 2 white balls and 3 black balls. Two balls are drawn and found to be one white and one black. Find the probability that the balls were drawn from the second bag.

\item Let
\[
a*b=a+b+ab
\]
Show that $*$ is commutative and associative.

\item Show that
\[
y=\frac{4\sin x}{2+\cos x}
\]
is increasing in
\[
\left[0,\frac{\pi}{2}\right]
\]

\item Using integration, find the area of triangle whose vertices are
\[
(1,-1),\ (2,-1),\ (1,-2)
\]

\item If
\[
A=
\begin{pmatrix}
1 & 0 & 2\\
0 & 2 & 1\\
2 & 0 & 3
\end{pmatrix},
\]
find
\[
A^{-1}
\]

\end{enumerate}
